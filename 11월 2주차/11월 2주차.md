
1. 퇴사

-난이도: 실버3

-완전탐색/그중 재귀를 이용한 dfs사용

- 해당 날짜의 상담을 선택하는경우, 안선택하는 경우를 모두 실행한 뒤 가장 최대이익인 값을 구한다.

```java
import java.util.Scanner;

public class 퇴사 {
	//퇴사
	//N+1일날 퇴사를 한다. 
	//남은날동안 얻을 수 있는 최대 수익을 구하시오
	//입력
	/*
	 * N(퇴사하기 전까지의 날/ 1-15) 
	 * N줄에 걸쳐 T(상담 소요 기간), P(상담 금액) -> 단 상담은 각각의 날에 시작할 수 있다.
	 * */
	//출력  최대 이익을 출력
	
	//N제한이 15로 작기때문에 완탐이 가능할것으로 예상
	static int N,max;
	static int[] days,price;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N=sc.nextInt();
		
		days=new int[N];
		price=new int[N];
		for (int i = 0; i < N; i++) {
			days[i]=sc.nextInt();
			price[i]=sc.nextInt();
		}
		max=Integer.MIN_VALUE;
		dfs(0,0,0);
		
		System.out.println(max);
		
	}

	public static void dfs(int day,int sumday,int sumprice) {
		if(day==N) {
			max=Math.max(sumprice, max);
			return;
		}
		
		if(day>=sumday&&day+days[day]<=N) {//해당 날짜의 상담을 선택 하는 경우
			dfs(day+1,day+days[day],sumprice+price[day]);
		}
		dfs(day+1,sumday,sumprice);//idx의 상담을 선택하지 않는 경우
	}
}
```

1. 스타트와 링크

-난이도:  실버 3

-완전탐색/그중 재귀를 이용한 dfs사용

- 팀을 뽑을때 boolean 배열을 사용해서 true인팀과 false인 팀으로 나눔

```java

import java.util.Arrays;
import java.util.Scanner;

public class 스타트와_링크 {
	//축구팀을 구성하는데 능력 차이를 최소화해서 팀을 나누기
	//능력치는 팀에 속한 모든 쌍의 능력치의 합
	//ex)1,2가 팀일때=> s[1][2]+s[2][1]
	//입력
	/*
	 * N(총 인원수), 항상 짝수이다.(4-20);
	 * N개의 줄에 걸쳐 N개의 S(능력치)
	 * */
	//출력  두팀간의 능력치의 차이의 최솟값
	static int N,s[][],ans;
	static boolean[] team;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N=sc.nextInt();
		s=new int[N][N];

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				s[i][j]=sc.nextInt();
			}
		}
	
		team=new boolean[N];
		ans=Integer.MAX_VALUE;
		dfs(0,0);
		
		System.out.println(ans);
	}
	
	 static void dfs(int idx, int cnt) {
	        if (cnt == (N / 2)) {
	            //차이값 계산
	        	int startS=0;
	        	int linkS=0;
	        	
	        	for (int i = 0; i < N-1; i++) {
					for (int j = i+1; j < N; j++) {
						if(team[i]&&team[j]) {
							startS+=s[i][j]+s[j][i];
						}else if(!team[i]&&!team[j]) {
							linkS+=s[i][j]+s[j][i];
						}
					}
				}
	        	
	        	int diff=Math.abs(startS-linkS);
	        	ans=Math.min(diff, ans);
	        	
	            return;
	        }

	        for (int i = idx; i < N; i++) {
	        	if(team[i]==false) {
	        		team[i]=true;
		            dfs(i + 1, cnt + 1);
		            team[i]=false;
	        	}
	        }
	    }

}
```
